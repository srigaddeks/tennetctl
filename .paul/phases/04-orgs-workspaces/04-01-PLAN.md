## Conventions
- 5-endpoint shape per sub-feature (GET list / POST create / GET one / PATCH one / DELETE soft-delete)
- PATCH handles ALL state changes (never POST /activate, /suspend, etc.)
- Response envelope: {"ok": true, "data": ...} or {"ok": false, "error": {...}}
- Routes acquire pool conn; pass conn to service; service uses conn (never pool)
- Audit emit after every mutation via `run_node("audit.events.emit", ctx, {event_key, outcome, metadata})`
- Soft-delete only (deleted_at = CURRENT_TIMESTAMP; returns 204)

    **`routes.py`** — FastAPI APIRouter:
    ```python
    router = APIRouter(prefix="/v1/orgs", tags=["iam.orgs"])

    def _build_ctx(request, audit_category="user"):
        # Extract x-user-id / x-session-id / x-org-id / x-workspace-id headers with defaults.
        # For v1 (no auth), default to audit_category='setup' for mutations, 'system' for reads,
        # with user_id='sys' / session='sys' etc. Phase 5+ replaces with JWT parse.
        ...
        return NodeContext(user_id=..., session_id=..., org_id=..., workspace_id=...,
                           audit_category=audit_category, trace_id=uuid7(), span_id=uuid7())

    @router.get("", status_code=200)
    async def list_orgs_route(request: Request, limit: int = 50, offset: int = 0):
        pool = request.app.state.pool
        async with pool.acquire() as conn:
            ctx = _build_ctx(request, audit_category="system")
            items, total = await service.list_orgs(conn, ctx, limit=limit, offset=offset)
            return response.success_list_response([OrgRead(**o).model_dump() for o in items],
                total=total, limit=limit, offset=offset)

    @router.post("", status_code=201)
    async def create_org_route(request: Request, body: OrgCreate):
        pool = request.app.state.pool
        async with pool.acquire() as conn:
            async with conn.transaction():
                ctx = _build_ctx(request, audit_category="setup")
                ctx = replace(ctx, conn=conn)
                org = await service.create_org(conn, ctx, slug=body.slug, display_name=body.display_name)
                return response.success_response(OrgRead(**org).model_dump())

    @router.get("/{org_id}", status_code=200)
    async def get_org_route(request: Request, org_id: str):
        pool = request.app.state.pool
        async with pool.acquire() as conn:
            ctx = _build_ctx(request, audit_category="system")
            org = await service.get_org(conn, ctx, org_id=org_id)
            if org is None:
                raise errors.NotFoundError(f"Org '{org_id}' not found.")
            return response.success_response(OrgRead(**org).model_dump())

    @router.patch("/{org_id}", status_code=200)
    async def update_org_route(request: Request, org_id: str, body: OrgUpdate):
        pool = request.app.state.pool
        async with pool.acquire() as conn:
            async with conn.transaction():
                ctx = _build_ctx(request, audit_category="setup")
                ctx = replace(ctx, conn=conn)
                org = await service.update_org(conn, ctx, org_id=org_id, patch=body)
                return response.success_response(OrgRead(**org).model_dump())

    @router.delete("/{org_id}", status_code=204)
    async def delete_org_route(request: Request, org_id: str):
        pool = request.app.state.pool
        async with pool.acquire() as conn:
            async with conn.transaction():
                ctx = _build_ctx(request, audit_category="setup")
                ctx = replace(ctx, conn=conn)
                await service.delete_org(conn, ctx, org_id=org_id)
        return Response(status_code=204)
    ```
