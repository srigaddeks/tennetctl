# Plan 11-03 — Subscriptions + Audit-Outbox Consumer + Critical Fan-Out

## Goal

Build the notify subscription pipeline:
1. `fct_notify_subscriptions` — maps event_key_pattern → template + channel
2. `fct_notify_deliveries` + `evt_notify_delivery_events` — delivery queue
3. Full CRUD API for subscriptions; read-only API for deliveries
4. Background worker: polls audit outbox → matches subscriptions → enqueues deliveries
5. Critical fan-out: category=critical → deliveries for email+webpush+in_app (not SMS)
6. 20+ pytest tests

```python
def matches_pattern(event_key: str, pattern: str) -> bool:
    """
    Exact: "iam.users.created" matches "iam.users.created"
    Suffix wildcard: "iam.users.*" matches "iam.users.created" and "iam.users.updated"
    Deep wildcard: "iam.*" matches any "iam.X" or "iam.X.Y"
    Global wildcard: "*" matches everything
    """
    if pattern == "*":
        return True
    if pattern == event_key:
        return True
    if pattern.endswith(".*"):
        prefix = pattern[:-2]  # strip ".*"
        return event_key.startswith(prefix + ".")
    return False
```

## Task 4: Worker + fan-out + manifest + routes mount

**worker.py** at `backend/02_features/06_notify/worker.py`:

```python
"""
Notify subscription worker.

Polls the audit outbox on 'audit_events' LISTEN channel (plus 30s fallback).
For each new audit event:
  1. Match all active subscriptions by org_id + event_key_pattern
  2. For each match: enqueue delivery rows (fan-out for critical category)

Worker state: cursor stored in-memory (resets to latest outbox ID on start).
"""

import asyncio
from importlib import import_module

_outbox_repo = import_module("backend.02_features.04_audit.sub_features.03_outbox.repository")
_sub_service = import_module("backend.02_features.06_notify.sub_features.05_subscriptions.service")
_del_service = import_module("backend.02_features.06_notify.sub_features.06_deliveries.service")
_tmpl_repo = import_module("backend.02_features.06_notify.sub_features.03_templates.repository")
_var_repo = import_module("backend.02_features.06_notify.sub_features.04_variables.repository")

_CRITICAL_CHANNELS = [1, 2, 3]  # email, webpush, in_app

async def process_audit_events(pool, since_id: int) -> int:
    """
    Poll outbox for events newer than since_id.
    Match active subscriptions, enqueue deliveries.
    Returns new cursor (last outbox_id processed, or since_id if nothing new).
    """
    async with pool.acquire() as conn:
        subscriptions = await _sub_service.list_active_for_worker(conn)
        rows = await _outbox_repo.poll_outbox(conn, since_id=since_id, limit=100)
        if not rows:
            return since_id
        for event in rows:
            for sub in subscriptions:
                if sub["org_id"] != event["org_id"]:
                    continue
                if not _sub_service.matches_pattern(event["event_key"], sub["event_key_pattern"]):
                    continue
                await _enqueue_for_subscription(conn, pool, sub, event)
        return rows[-1]["outbox_id"]


async def _enqueue_for_subscription(conn, pool, subscription, audit_event):
    template = await _tmpl_repo.get_template(conn, template_id=subscription["template_id"])
    if template is None:
        return
    is_critical = template.get("category_code") == "critical"
    channels = _CRITICAL_CHANNELS if is_critical else [subscription["channel_id"]]
    context = {
        "actor_user_id": audit_event.get("actor_user_id"),
        "org_id": audit_event.get("org_id"),
        "workspace_id": audit_event.get("workspace_id"),
        "event_metadata": audit_event.get("metadata"),
    }
    resolved = await _var_repo.resolve_variables(conn, template_id=template["id"], context=context)
    priority_id = template.get("priority_id") or 2  # default normal
    for channel_id in channels:
        await _del_service.create_delivery(
            conn,
            subscription_id=subscription["id"],
            org_id=subscription["org_id"],
            template_id=template["id"],
            recipient_user_id=audit_event.get("actor_user_id", ""),
            channel_id=channel_id,
            priority_id=priority_id,
            resolved_variables=resolved,
            audit_outbox_id=audit_event.get("outbox_id"),
        )


async def _worker_loop(pool):
    """Background loop: LISTEN + 30s poll fallback."""
    async with pool.acquire() as conn:
        since_id = await _outbox_repo.latest_outbox_id(conn)
    
    _wake = asyncio.Event()
    
    async def _listener(conn, pid, channel, payload):
        _wake.set()
    
    listen_conn = await pool.acquire()
    await listen_conn.add_listener("audit_events", _listener)
    try:
        while True:
            try:
                await asyncio.wait_for(_wake.wait(), timeout=30.0)
            except asyncio.TimeoutError:
                pass
            _wake.clear()
            since_id = await process_audit_events(pool, since_id)
    finally:
        await listen_conn.remove_listener("audit_events", _listener)
        await pool.release(listen_conn)


def start_worker(pool) -> asyncio.Task:
    """Call from app lifespan to start the background worker task."""
    return asyncio.create_task(_worker_loop(pool))
```
